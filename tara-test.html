import { db } from "./firebase.js"; import { getAuth, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js"; document.addEventListener("DOMContentLoaded", () => { // ================= AUTH ================= const auth = getAuth(); onAuthStateChanged(auth, (user) => { const topbarUser = document.getElementById("navUsername"); if (user && topbarUser) { const name = user.email.split("@")[0]; topbarUser.textContent = name.toUpperCase(); } }); // ================= LOAD PROJECTS ================= loadProjects(); }); // BUTTON EVENTS const newProjectBtn = document.getElementById("newProjectBtn"); const joinProjectBtn = document.getElementById("joinProjectBtn"); const formContainer = document.getElementById("teamFormContainer"); newProjectBtn.addEventListener("click", showCreateForm); joinProjectBtn.addEventListener("click", showJoinForm); // CREATE PROJECT FORM function showCreateForm() { formContainer.innerHTML = `

Create New Project

`; document.getElementById("cancelBtn").addEventListener("click", clearForm); document.getElementById("createBtn").addEventListener("click", createProject); } // JOIN FORM function showJoinForm() { formContainer.innerHTML = `

Join a Project

`; document.getElementById("cancelBtn").addEventListener("click", clearForm); document.getElementById("joinBtn").addEventListener("click", joinProject); } // CLEAR FORM function clearForm() { formContainer.innerHTML = ""; } // CREATE PROJECT LOGIC import { collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js"; async function createProject() { const name = document.getElementById("projectName").value.trim(); const desc = document.getElementById("projectDesc").value.trim(); if (!name) { alert("Project name required"); return; } const code = generateCode(); const auth = getAuth(); const user = auth.currentUser; try { await addDoc(collection(db, "projects"), { name, desc, code, members: 1, owner: user.email, createdAt: serverTimestamp() }); clearForm(); } catch (err) { console.error("PROJECT CREATE ERROR:", err); } } // RENDER PROJECT CARD function renderProject(project) { const list = document.getElementById("projectList"); const div = document.createElement("div"); div.className = "project-card"; const roleBadge = project.role === "admin" ? `
Admin
` : ""; div.innerHTML = ` ${roleBadge}

${project.name}

${project.desc}

Code: ${project.code}
${project.members}
`; list.appendChild(div); } // COPY FUNCTINALITY document.addEventListener("click", (e) => { if (e.target.classList.contains("copy-icon")) { const icon = e.target; const code = icon.dataset.code; navigator.clipboard.writeText(code); // 🔥 ANIMATION TRIGGER icon.classList.add("copied"); setTimeout(() => { icon.classList.remove("copied"); }, 300); } }); // EMPTY STATE CONTROL function hideEmptyState() { const empty = document.getElementById("emptyProject"); if (empty) { empty.style.display = "none"; empty.remove(); // 🔥 removes it completely (bulletproof) } } // GENERATE INVITE CODE function generateCode() { return Math.random().toString(36).substring(2, 8).toUpperCase(); } // LOAD PROJECT AFTER REFRESH import { onSnapshot, query, orderBy } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js"; function loadProjects() { const list = document.getElementById("projectList"); const q = query(collection(db, "projects"), orderBy("createdAt", "desc")); onSnapshot(q, (snapshot) => { list.innerHTML = ""; // clear UI if (snapshot.empty) { showEmptyState(); return; } hideEmptyState(); snapshot.docs.forEach(doc => { const project = doc.data(); renderProject({ ...project, role: project.owner === getAuth().currentUser?.email ? "admin" : "member" }); }); }); } function showEmptyState() { const empty = document.getElementById("emptyProject"); if (empty) empty.style.display = "flex"; } // ACTIVITY BUTTON CLICKER const activityBtn = document.getElementById("activityBtn"); activityBtn.addEventListener("click", () => { activityBtn.style.transform = "scale(0.95)"; setTimeout(() => { window.location.href = "./activity.html"; }, 100); });
